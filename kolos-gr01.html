4. Letters

4.1. Zadanie

# zmień poniższy kod żeby zamiast
# index: 0 element: a | a
# index: 1 element: b | bb
# index: 2 element: c | ccc
# index: 3 element: d | dddd
# index: 4 element: e | eeeee
# index: 5 element: f | ffffff
# index: 6 element: g | ggggggg
# index: 7 element: h | hhhhhhhh
#
# rysował
#
# index: 0 element: h | h
# index: 1 element: g | gg
# index: 2 element: f | fff
# index: 3 element: e | eeee
# index: 4 element: d | ddddd
# index: 5 element: c | cccccc
# index: 6 element: b | bbbbbbb
# index: 7 element: a | aaaaaaaa

from string import ascii_lowercase as alpha
total = 8
letters = alpha[:total]
for idx, letter in enumerate(letters):
    print(
        f"index: {idx} element: {letter} |" +
        f" {letter * (idx + 1)}")

print("OK")
